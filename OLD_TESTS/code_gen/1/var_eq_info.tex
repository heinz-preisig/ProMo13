	\Large{Equations}
	\begin{multicols}{2}
	\begin{enumerate}
		\item $n_{NS} = \int_{t_o}^{t_e} \dot{n}_{NS} dt + n^{o}_{NS}$
		\item $\dot{n}_{NS} = \hat{n}^{c}_{NS} + \hat{n}^{d}_{NS} + \tilde{n}_{NS}$
		\item $\hat{n}^{c}_{NS} = F^{c}_{AS,NS} \overset{AS}{\star} \hat{n}^{c}_{AS}$
		\item $\hat{n}^{c}_{AS} = \hat{V}_{A} \odot c_{AS}$
		\item $\hat{V}_{A} = cv_A D^{c}_{N,A} \overset{N}{\star} p_N$
		\item $c_{AS} = \dfrac{1}{2} \left[ D^{c}_{NS,AS} - d_A \odot \left|D^{c}_{NS,AS} \right|  \right] \overset{NS}{\star} c_{NS}$
		\item $\hat{E}^{v}_{A} = -\hat{V}_{A} P_{N,A} \overset{N}{\star} p_N$
		\item $d_A = Sign(D^{c}_{N,A} \overset{N}{\star} p_N)$
		\item $c_{NS} = \dfrac{n_{NS}}{V_N}$
		\item $\dot{E}^{v}_N = F_{N,A} \overset{A}{\star} \hat{E}^{v}_{A}$
		\item $V_N = \dfrac{m_N}{\rho_N}$
		\item $\dot{0} = \dot{E}^{v}_N + \dot{E}^{p}_N + \dot{E}^{k}_N + \dot{E}^{f}_N$
		\item $m_N = \lambda_S \overset{S \in NS}{\star} n_{NS}$
		\item $\dot{E}^{p}_N = F_{N,A} \overset{A}{\star} \hat{E}^{p}_{A}$
		\item $\dot{E}^{k}_N = F_{N,A} \overset{A}{\star} \hat{E}^{k}_{A}$
		\item $\dot{E}^{f}_N = k^{f}_A l_A$
		\item $\hat{E}^{p}_{A} = \hat{m}_A g h_A$
		\item $\hat{E}^{k}_{A} = \dfrac{1}{2} \hat{m}_A v_A^2$
		\item $\hat{m}_A = \rho_A \hat{V}_{A}$
		\item $v_A = \dfrac{\hat{V}_{A}}{A_A} $
	\end{enumerate}
